Questions sum of multiples of 3 and 15 < N
THESE CAN BE REPRESENTED IN THE FORM OF AP 

```
void solve(){
    
    /*
        Sum 
        for 3 ] 
        3,6,9,12,....n/3 * 3

        for 5]
        5,10,15,20,...n/5 * 5

        for 15]
        15,30,45,....n/15 * 15
    */
    int n; cin>>n;
    auto sn = [&](auto x)  {return ((n-1)/x)* (x + ((n-1)/x)*x) * 1/2;};

    int s3 = sn(3);
    int s5 = sn(5);
    int s15 = sn(15);
    debug(s3,s5,s15);
    cout<<s3+s5-s15<<endl;

}
```
